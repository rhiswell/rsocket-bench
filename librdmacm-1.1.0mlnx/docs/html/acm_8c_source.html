130 void ucma_ib_init(void)
131 {
132  struct sockaddr_in addr;
133  static int init;
134  int ret;
135 
136  if (init)
137  return;
138 
139  pthread_mutex_lock(&acm_lock);
140  if (init)
141  goto unlock;
142 
143  if (!ucma_set_server_port())
144  goto out;
145 
146  sock = socket(AF_INET, SOCK_STREAM | SOCK_CLOEXEC, IPPROTO_TCP);
147  if (sock < 0)
148  goto out;
149 
150  memset(&addr, 0, sizeof addr);
151  addr.sin_family = AF_INET;
152  addr.sin_addr.s_addr = htonl(INADDR_LOOPBACK);
153  addr.sin_port = htons(server_port);
154  ret = connect(sock, (struct sockaddr *) &addr, sizeof(addr));
155  if (ret) {
156  close(sock);
157  sock = -1;
158  }
159 out:
160  init = 1;
161 unlock:
162  pthread_mutex_unlock(&acm_lock);
163 }

173 static int ucma_ib_set_addr(struct rdma_addrinfo *ib_rai,
174  struct rdma_addrinfo *rai)
175 {
176  struct sockaddr_ib *src, *dst;
177  struct ibv_path_record *path;
178 
179  src = calloc(1, sizeof *src);
180  if (!src)
181  return ERR(ENOMEM);
182 
183  dst = calloc(1, sizeof *dst);
184  if (!dst) {
185  free(src);
186  return ERR(ENOMEM);
187  }
188 
189  path = &((struct ibv_path_data *) ib_rai->ai_route)->path;
190 
191  src->sib_family = AF_IB;
192  src->sib_pkey = path->pkey;
193  src->sib_flowinfo = htonl(ntohl(path->flowlabel_hoplimit) >> 8);
194  memcpy(&src->sib_addr, &path->sgid, 16);
195  ucma_set_sid(ib_rai->ai_port_space, rai->ai_src_addr, src);
196 
197  dst->sib_family = AF_IB;
198  dst->sib_pkey = path->pkey;
199  dst->sib_flowinfo = htonl(ntohl(path->flowlabel_hoplimit) >> 8);
200  memcpy(&dst->sib_addr, &path->dgid, 16);
201  ucma_set_sid(ib_rai->ai_port_space, rai->ai_dst_addr, dst);
202 
203  ib_rai->ai_src_addr = (struct sockaddr *) src;
204  ib_rai->ai_src_len = sizeof(*src);
205 
206  ib_rai->ai_dst_addr = (struct sockaddr *) dst;
207  ib_rai->ai_dst_len = sizeof(*dst);
208 
209  return 0;
210 }

289 static void ucma_ib_save_resp(struct rdma_addrinfo *rai, struct acm_msg *msg)
290 {
291  struct acm_ep_addr_data *ep_data;
292  struct ibv_path_data *path_data = NULL;
293  struct sockaddr_in *sin;
294  struct sockaddr_in6 *sin6;
295  int i, cnt, path_cnt = 0;
296 
297  cnt = (msg->hdr.length - ACM_MSG_HDR_LENGTH) / ACM_MSG_EP_LENGTH;
298  for (i = 0; i < cnt; i++) {
299  ep_data = &msg->resolve_data[i];
300  switch (ep_data->type) {
301  case ACM_EP_INFO_PATH:
302  ep_data->type = 0;
303  if (!path_data)
304  path_data = (struct ibv_path_data *) ep_data;
305  path_cnt++;
306  break;
307  case ACM_EP_INFO_ADDRESS_IP:
308  if (!(ep_data->flags & ACM_EP_FLAG_SOURCE) || rai->ai_src_len)
309  break;
310 
311  sin = calloc(1, sizeof(*sin));
312  if (!sin)
313  break;
314 
315  sin->sin_family = AF_INET;
316  memcpy(&sin->sin_addr, &ep_data->info.addr, 4);
317  rai->ai_src_len = sizeof(*sin);
318  rai->ai_src_addr = (struct sockaddr *) sin;
319  break;
320  case ACM_EP_INFO_ADDRESS_IP6:
321  if (!(ep_data->flags & ACM_EP_FLAG_SOURCE) || rai->ai_src_len)
322  break;
323 
324  sin6 = calloc(1, sizeof(*sin6));
325  if (!sin6)
326  break;
327 
328  sin6->sin6_family = AF_INET6;
329  memcpy(&sin6->sin6_addr, &ep_data->info.addr, 16);
330  rai->ai_src_len = sizeof(*sin6);
331  rai->ai_src_addr = (struct sockaddr *) sin6;
332  break;
333  default:
334  break;
335  }
336  }
337 
338  rai->ai_route = calloc(path_cnt, sizeof(*path_data));
339  if (rai->ai_route) {
340  memcpy(rai->ai_route, path_data, path_cnt * sizeof(*path_data));
341  rai->ai_route_len = path_cnt * sizeof(*path_data);
342  }
343 }

367 void ucma_ib_resolve(struct rdma_addrinfo **rai, struct rdma_addrinfo *hints)
368 {
369  struct acm_msg msg;
370  struct acm_ep_addr_data *data;
371  int ret;
372 
373  ucma_ib_init();
374  if (sock < 0)
375  return;
376 
377  memset(&msg, 0, sizeof msg);
378  msg.hdr.version = ACM_VERSION;
379  msg.hdr.opcode = ACM_OP_RESOLVE;
380  msg.hdr.length = ACM_MSG_HDR_LENGTH;
381 
382  data = &msg.resolve_data[0];
383  if (ucma_inet_addr((*rai)->ai_src_addr, (*rai)->ai_src_len)) {
384  data->flags = ACM_EP_FLAG_SOURCE;
385  ucma_set_ep_addr(data, (*rai)->ai_src_addr);
386  data++;
387  msg.hdr.length += ACM_MSG_EP_LENGTH;
388  }
389 
390  if (ucma_inet_addr((*rai)->ai_dst_addr, (*rai)->ai_dst_len)) {
391  data->flags = ACM_EP_FLAG_DEST;
392  if (hints->ai_flags & (RAI_NUMERICHOST | RAI_NOROUTE))
393  data->flags |= ACM_FLAGS_NODELAY;
394  ucma_set_ep_addr(data, (*rai)->ai_dst_addr);
395  data++;
396  msg.hdr.length += ACM_MSG_EP_LENGTH;
397  }
398 
399  if (hints->ai_route_len ||
400  ucma_ib_addr((*rai)->ai_src_addr, (*rai)->ai_src_len) ||
401  ucma_ib_addr((*rai)->ai_dst_addr, (*rai)->ai_dst_len)) {
402  struct ibv_path_record *path;
403 
404  if (hints->ai_route_len == sizeof(struct ibv_path_record))
405  path = (struct ibv_path_record *) hints->ai_route;
406  else if (hints->ai_route_len == sizeof(struct ibv_path_data))
407  path = &((struct ibv_path_data *) hints->ai_route)->path;
408  else
409  path = NULL;
410 
411  if (path)
412  memcpy(&data->info.path, path, sizeof(*path));
413 
414  if (ucma_ib_addr((*rai)->ai_src_addr, (*rai)->ai_src_len)) {
415  memcpy(&data->info.path.sgid,
416  &((struct sockaddr_ib *) (*rai)->ai_src_addr)->sib_addr, 16);
417  }
418  if (ucma_ib_addr((*rai)->ai_dst_addr, (*rai)->ai_dst_len)) {
419  memcpy(&data->info.path.dgid,
420  &((struct sockaddr_ib *) (*rai)->ai_dst_addr)->sib_addr, 16);
421  }
422  data->type = ACM_EP_INFO_PATH;
423  data++;
424  msg.hdr.length += ACM_MSG_EP_LENGTH;
425  }
426 
427  pthread_mutex_lock(&acm_lock);
428  ret = send(sock, (char *) &msg, msg.hdr.length, 0);
429  if (ret != msg.hdr.length) {
430  pthread_mutex_unlock(&acm_lock);
431  return;
432  }
433 
434  ret = recv(sock, (char *) &msg, sizeof msg, 0);
435  pthread_mutex_unlock(&acm_lock);
436  if (ret < ACM_MSG_HDR_LENGTH || ret != msg.hdr.length || msg.hdr.status)
437  return;
438 
439  ucma_ib_save_resp(*rai, &msg);
440 
441  if (af_ib_support && !(hints->ai_flags & RAI_ROUTEONLY) && (*rai)->ai_route_len)
442  ucma_resolve_af_ib(rai);
443 }
