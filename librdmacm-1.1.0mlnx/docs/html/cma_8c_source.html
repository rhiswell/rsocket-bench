139 static void ucma_cleanup(void)
140 {
141  ucma_ib_cleanup();
142 
143  if (cma_dev_cnt) {
144  while (cma_dev_cnt--) {
145  if (!cma_dev_array[cma_dev_cnt].verbs)
146  continue;

161 static int check_abi_version(void)
162 {
163  char value[8];
164 
165  if ((ibv_read_sysfs_file(ibv_get_sysfs_path(),
166  "class/misc/rdma_cm/abi_version",
167  value, sizeof value) < 0) &&
168  (ibv_read_sysfs_file(ibv_get_sysfs_path(),
169  "class/infiniband_ucma/abi_version",
170  value, sizeof value) < 0)) {
171  /*
172  * Older version of Linux do not have class/misc. To support
173  * backports, assume the most recent version of the ABI. If
174  * we're wrong, we'll simply fail later when calling the ABI.
175  */
176  return 0;
177  }
178 
179  abi_ver = strtol(value, NULL, 10);
180  if (abi_ver < RDMA_USER_CM_MIN_ABI_VERSION ||
181  abi_ver > RDMA_USER_CM_MAX_ABI_VERSION) {
182  return -1;
183  }
184  return 0;
185 }

213 int ucma_init(void)
214 {
215  struct ibv_device **dev_list = NULL;
216  int i, ret, dev_cnt;
217 
218  /* Quick check without lock to see if we're already initialized */
219  if (cma_dev_cnt)
220  return 0;
221 
222  pthread_mutex_lock(&mut);
223  if (cma_dev_cnt) {
224  pthread_mutex_unlock(&mut);
225  return 0;
226  }
227 
228  fastlock_init(&idm_lock);
229  ret = check_abi_version();
230  if (ret)
231  goto err1;
232 
233  dev_list = ibv_get_device_list(&dev_cnt);
234  if (!dev_list) {
235  ret = ERR(ENODEV);
236  goto err1;
237  }
238 
239  if (!dev_cnt) {
240  ret = ERR(ENODEV);
241  goto err2;
242  }
243 
244  cma_dev_array = calloc(dev_cnt, sizeof *cma_dev_array);
245  if (!cma_dev_array) {
246  ret = ERR(ENOMEM);
247  goto err2;
248  }
249 
250  for (i = 0; dev_list[i]; i++)
251  cma_dev_array[i].guid = ibv_get_device_guid(dev_list[i]);
252 
253  cma_dev_cnt = dev_cnt;
254  ucma_set_af_ib_support();
255  pthread_mutex_unlock(&mut);
256  ibv_free_device_list(dev_list);
257  return 0;
258 
259 err2:
260  ibv_free_device_list(dev_list);
261 err1:
262  fastlock_destroy(&idm_lock);
263  pthread_mutex_unlock(&mut);
264  return ret;
265 }

289 static int ucma_init_device(struct cma_device *cma_dev)
290 {
291  struct ibv_port_attr port_attr;
292  struct ibv_device_attr attr;
293  int i, ret;
294 
295  if (cma_dev->verbs)
296  return 0;
297 
298  cma_dev->verbs = ucma_open_device(cma_dev->guid);
299  if (!cma_dev->verbs)
300  return ERR(ENODEV);
301 
302  ret = ibv_query_device(cma_dev->verbs, &attr);
303  if (ret) {
304  ret = ERR(ret);
305  goto err;
306  }
307 
308  cma_dev->port = malloc(sizeof *cma_dev->port * attr.phys_port_cnt);
309  if (!cma_dev->port) {
310  ret = ERR(ENOMEM);
311  goto err;
312  }
313 
314  for (i = 1; i <= attr.phys_port_cnt; i++) {
315  if (ibv_query_port(cma_dev->verbs, i, &port_attr))
316  cma_dev->port[i - 1].link_layer = IBV_LINK_LAYER_UNSPECIFIED;
317  else
318  cma_dev->port[i - 1].link_layer = port_attr.link_layer;
319  }
320 
321  cma_dev->port_cnt = attr.phys_port_cnt;
322  cma_dev->max_qpsize = attr.max_qp_wr;
323  cma_dev->max_initiator_depth = (uint8_t) attr.max_qp_init_rd_atom;
324  cma_dev->max_responder_resources = (uint8_t) attr.max_qp_rd_atom;
325  cma_init_cnt++;
326  return 0;
327 
328 err:
329  ibv_close_device(cma_dev->verbs);
330  cma_dev->verbs = NULL;
331  return ret;
332 }

334 static int ucma_init_all(void)
335 {
336  int i, ret = 0;
337 
338  if (!cma_dev_cnt) {
339  ret = ucma_init();
340  if (ret)
341  return ret;
342  }
343 
344  if (cma_init_cnt == cma_dev_cnt)
345  return 0;
346 
347  pthread_mutex_lock(&mut);

349  ret = ucma_init_device(&cma_dev_array[i]);
350  if (ret)
351  break;
352  }
353  pthread_mutex_unlock(&mut);
354  return ret;

424 static int ucma_get_device(struct cma_id_private *id_priv, uint64_t guid)
425 {
426  struct cma_device *cma_dev;
427  int i, ret;
428 
429  for (i = 0; i < cma_dev_cnt; i++) {
430  cma_dev = &cma_dev_array[i];
431  if (cma_dev->guid == guid)
432  goto match;
433  }
434 
435  return ERR(ENODEV);
436 match:
437  pthread_mutex_lock(&mut);
438  if ((ret = ucma_init_device(cma_dev)))

447  }
448  }
449  id_priv->cma_dev = cma_dev;
450  id_priv->id.verbs = cma_dev->verbs;
451  id_priv->id.pd = cma_dev->pd;
452 out:
453  pthread_mutex_unlock(&mut);
454  return ret;
455 }
