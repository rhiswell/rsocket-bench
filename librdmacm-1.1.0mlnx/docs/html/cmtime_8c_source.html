117 {

156 static inline void list_add_tail(struct work_list *work_list, struct list_head *req)
157 {
158  int empty;
159  pthread_mutex_lock(&work_list->lock);
160  empty = __list_empty(work_list);
161  req->prev = work_list->list.prev;
162  req->next = &work_list->list;
163  req->prev->next = work_list->list.prev = req;
164  pthread_mutex_unlock(&work_list->lock);
165  if (empty)
166  pthread_cond_signal(&work_list->cond);
167 }

179 static void show_perf(void)
180 {
181  int c, i;
182  float us, max[STEP_CNT], min[STEP_CNT];
183 
184  for (i = 0; i < STEP_CNT; i++) {
185  max[i] = 0;
186  min[i] = 999999999.;
187  for (c = 0; c < connections; c++) {
188  if (!zero_time(&nodes[c].times[i][0]) &&
189  !zero_time(&nodes[c].times[i][1])) {
190  us = diff_us(&nodes[c].times[i][1], &nodes[c].times[i][0]);
191  if (us > max[i])
192  max[i] = us;
193  if (us < min[i])
194  min[i] = us;
195  }
196  }
197  }
198 
199  printf("step total ms max ms min us us / conn\n");
200  for (i = 0; i < STEP_CNT; i++) {
201  if (i == STEP_BIND && !src_addr)
202  continue;
203 
204  us = diff_us(&times[i][1], &times[i][0]);
205  printf("%-13s: %11.2f%11.2f%11.2f%11.2f\n", step_str[i], us / 1000.,
206  max[i] / 1000., min[i], us / connections);
207  }
208 }

259 {

289 static void cma_handler(struct rdma_cm_id *id, struct rdma_cm_event *event)
290 {
291  struct node *n = id->context;
292  struct list_head *request;
293 
294  switch (event->event) {
295  case RDMA_CM_EVENT_ADDR_RESOLVED:
296  addr_handler(n);
297  break;
298  case RDMA_CM_EVENT_ROUTE_RESOLVED:
299  route_handler(n);
300  break;
301  case RDMA_CM_EVENT_CONNECT_REQUEST:
302  request = malloc(sizeof *request);
303  if (!request) {
304  perror("out of memory accepting connect request");
305  rdma_reject(id, NULL, 0);
306  rdma_destroy_id(id);
307  } else {
308  INIT_LIST(request);
309  request->id = id;
310  list_add_tail(&req_work, request);
311  }
312  break;
313  case RDMA_CM_EVENT_ESTABLISHED:
314  if (n)
315  conn_handler(n);
316  break;
317  case RDMA_CM_EVENT_ADDR_ERROR:
318  if (n->retries--) {
319  if (!rdma_resolve_addr(n->id, rai->ai_src_addr,
320  rai->ai_dst_addr, timeout))
321  break;
322  }
323  printf("RDMA_CM_EVENT_ADDR_ERROR, error: %d\n", event->status);
324  addr_handler(n);
325  n->error = 1;
326  break;
327  case RDMA_CM_EVENT_ROUTE_ERROR:
328  if (n->retries--) {
329  if (!rdma_resolve_route(n->id, timeout))
330  break;
331  }
332  printf("RDMA_CM_EVENT_ROUTE_ERROR, error: %d\n", event->status);
333  route_handler(n);
334  n->error = 1;
335  break;
336  case RDMA_CM_EVENT_CONNECT_ERROR:
337  case RDMA_CM_EVENT_UNREACHABLE:
338  case RDMA_CM_EVENT_REJECTED:
339  printf("event: %s, error: %d\n",
340  rdma_event_str(event->event), event->status);
341  conn_handler(n);
342  n->error = 1;
343  break;
344  case RDMA_CM_EVENT_DISCONNECTED:
345  if (!n) {
346  request = malloc(sizeof *request);
347  if (!request) {
348  perror("out of memory queueing disconnect request, handling synchronously");
349  rdma_disconnect(id);
350  rdma_destroy_id(id);
351  } else {
352  INIT_LIST(request);
353  request->id = id;
354  list_add_tail(&disc_work, request);
355  }
356  } else
357  disc_handler(n);
358  break;
359  case RDMA_CM_EVENT_DEVICE_REMOVAL:
360  /* Cleanup will occur after test completes. */
361  break;
362  default:
363  break;
364  }
365  rdma_ack_cm_event(event);
366 }

430 static int run_server(void)
431 {
432  pthread_t req_thread, disc_thread;
433  struct rdma_cm_id *listen_id;
434  int ret;
435 
436  INIT_LIST(&req_work.list);
437  INIT_LIST(&disc_work.list);
438  ret = pthread_mutex_init(&req_work.lock, NULL);
439  if (ret) {
440  perror("initializing mutex for req work");
441  return ret;
442  }
443 
444  ret = pthread_mutex_init(&disc_work.lock, NULL);
445  if (ret) {
446  perror("initializing mutex for disc work");
447  return ret;
448  }
449 
450  ret = pthread_cond_init(&req_work.cond, NULL);
451  if (ret) {
452  perror("initializing cond for req work");
453  return ret;
454  }
455 
456  ret = pthread_cond_init(&disc_work.cond, NULL);
457  if (ret) {
458  perror("initializing cond for disc work");
459  return ret;
460  }
461 
462  ret = pthread_create(&req_thread, NULL, req_handler_thread, NULL);
463  if (ret) {
464  perror("failed to create req handler thread");
465  return ret;
466  }
467 
468  ret = pthread_create(&disc_thread, NULL, disc_handler_thread, NULL);
469  if (ret) {
470  perror("failed to create disconnect handler thread");
471  return ret;
472  }
473 
474  ret = rdma_create_id(channel, &listen_id, NULL, hints.ai_port_space);
475  if (ret) {
476  perror("listen request failed");
477  return ret;
478  }
479 
480  ret = get_rdma_addr(src_addr, dst_addr, port, &hints, &rai);
481  if (ret) {
482  perror("getrdmaaddr error");
483  goto out;
484  }
485 
486  ret = rdma_bind_addr(listen_id, rai->ai_src_addr);
487  if (ret) {
488  perror("bind address failed");
489  goto out;
490  }
491 
492  ret = rdma_listen(listen_id, 0);
493  if (ret) {
494  perror("failure trying to listen");
495  goto out;
496  }
497 
498  process_events(NULL);
499  out:
500  rdma_destroy_id(listen_id);
501  return ret;
502 }

624 
625  return ret;
626 }
