44 /*
45  * Indexer - to find a structure given an index
46  *
47  * We store pointers using a double lookup and return an index to the
48  * user which is then used to retrieve the pointer. The upper bits of
49  * the index are itself an index into an array of memory allocations.
50  * The lower bits specify the offset into the allocated memory where
51  * the pointer is stored.
52  *
53  * This allows us to adjust the number of pointers stored by the index
54  * list without taking a lock during data lookups.
55  */

57 static int idx_grow(struct indexer *idx)
58 {
59  union idx_entry *entry;
60  int i, start_index;
61 
62  if (idx->size >= IDX_ARRAY_SIZE)
63  goto nomem;
64 
65  idx->array[idx->size] = calloc(IDX_ENTRY_SIZE, sizeof(union idx_entry));
66  if (!idx->array[idx->size])
67  goto nomem;
68 
69  entry = idx->array[idx->size];
70  start_index = idx->size << IDX_ENTRY_BITS;
71  entry[IDX_ENTRY_SIZE - 1].next = idx->free_list;
72 
73  for (i = IDX_ENTRY_SIZE - 2; i >= 0; i--)
74  entry[i].next = start_index + i + 1;
75 
76  /* Index 0 is reserved */
77  if (start_index == 0)
78  start_index++;
79  idx->free_list = start_index;
80  idx->size++;
81  return start_index;
82 
83 nomem:
84  errno = ENOMEM;
85  return -1;
86 }

88 int idx_insert(struct indexer *idx, void *item)
89 {

91  int index;
92 
93  if ((index = idx->free_list) == 0) {

100  entry[idx_entry_index(index)].item = item;
101  return index;
