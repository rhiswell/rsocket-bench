43 /*
44  * Indexer - to find a structure given an index. Synchronization
45  * must be provided by the caller. Caller must initialize the
46  * indexer by setting free_list and size to 0.
47  */
48 
49 union idx_entry {
50  void *item;

52 };

54 #define IDX_INDEX_BITS 16
55 #define IDX_ENTRY_BITS 10
56 #define IDX_ENTRY_SIZE (1 << IDX_ENTRY_BITS)
57 #define IDX_ARRAY_SIZE (1 << (IDX_INDEX_BITS - IDX_ENTRY_BITS))
58 #define IDX_MAX_INDEX ((1 << IDX_INDEX_BITS) - 1)
