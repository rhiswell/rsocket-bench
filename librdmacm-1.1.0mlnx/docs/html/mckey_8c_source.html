87 static int create_message(struct cmatest_node *node)
88 {
89  if (!message_size)
90  message_count = 0;
91 
92  if (!message_count)
93  return 0;
94 
95  node->mem = malloc(message_size + sizeof(struct ibv_grh));
96  if (!node->mem) {
97  printf("failed message allocation\n");
98  return -1;
99  }
100  node->mr = ibv_reg_mr(node->pd, node->mem,
101  message_size + sizeof(struct ibv_grh),
102  IBV_ACCESS_LOCAL_WRITE);
103  if (!node->mr) {
104  printf("failed to reg MR\n");
105  goto err;
106  }
107  return 0;
108 err:
109  free(node->mem);
110  return -1;
111 }

113 static int verify_test_params(struct cmatest_node *node)
114 {
115  struct ibv_port_attr port_attr;
116  int ret;
117 
118  ret = ibv_query_port(node->cma_id->verbs, node->cma_id->port_num,
119  &port_attr);
120  if (ret)
121  return ret;
122 
123  if (message_count && message_size > (1 << (port_attr.active_mtu + 7))) {
124  printf("mckey: message_size %d is larger than active mtu %d\n",

177 static int post_recvs(struct cmatest_node *node)
178 {
179  struct ibv_recv_wr recv_wr, *recv_failure;
180  struct ibv_sge sge;
181  int i, ret = 0;
182 
183  if (!message_count)
184  return 0;
185 
186  recv_wr.next = NULL;
187  recv_wr.sg_list = &sge;
188  recv_wr.num_sge = 1;
189  recv_wr.wr_id = (uintptr_t) node;

191  sge.length = message_size + sizeof(struct ibv_grh);
192  sge.lkey = node->mr->lkey;
193  sge.addr = (uintptr_t) node->mem;
194 
195  for (i = 0; i < message_count && !ret; i++ ) {
196  ret = ibv_post_recv(node->cma_id->qp, &recv_wr, &recv_failure);
197  if (ret) {

199  break;

201  }

218  send_wr.send_flags = signal_flag;

229

297 static int cma_handler(struct rdma_cm_id *cma_id, struct rdma_cm_event *event)
298 {
299  int ret = 0;
300 
301  switch (event->event) {
302  case RDMA_CM_EVENT_ADDR_RESOLVED:
303  ret = addr_handler(cma_id->context);
304  break;
305  case RDMA_CM_EVENT_MULTICAST_JOIN:
306  ret = join_handler(cma_id->context, &event->param.ud);
307  break;
308  case RDMA_CM_EVENT_ADDR_ERROR:
309  case RDMA_CM_EVENT_ROUTE_ERROR:
310  case RDMA_CM_EVENT_MULTICAST_ERROR:
311  printf("mckey: event: %s, error: %d\n",
312  rdma_event_str(event->event), event->status);
313  connect_error();
314  ret = event->status;
315  break;
316  case RDMA_CM_EVENT_DEVICE_REMOVAL:
317  /* Cleanup will occur after test completes. */
318  break;
319  default:
320  break;
321  }
322  return ret;
323 }

478 static int run(void)
479 {
480  int i, ret;
481 
482  printf("mckey: starting %s\n", is_sender ? "client" : "server");
483  if (src_addr) {
484  ret = get_addr(src_addr, (struct sockaddr *) &test.src_in);
485  if (ret)
486  return ret;
487  }
488 
489  ret = get_dst_addr(dst_addr, (struct sockaddr *) &test.dst_in);
490  if (ret)
491  return ret;
492 
493  printf("mckey: joining\n");
494  for (i = 0; i < connections; i++) {
495  if (src_addr) {
496  ret = rdma_bind_addr(test.nodes[i].cma_id,
497  test.src_addr);
498  if (ret) {
499  perror("mckey: addr bind failure");
500  connect_error();
501  return ret;
502  }
503  }
504 
505  if (unmapped_addr)
506  ret = addr_handler(&test.nodes[i]);
507  else
508  ret = rdma_resolve_addr(test.nodes[i].cma_id,
509  test.src_addr, test.dst_addr,
510  2000);
511  if (ret) {
512  perror("mckey: resolve addr failure");
513  connect_error();
514  return ret;
515  }
516  }
517 
518  ret = connect_events();
519  if (ret)
520  goto out;
521 
522  pthread_create(&test.cmathread, NULL, cma_thread, NULL);
523 
524  /*
525  * Pause to give SM chance to configure switches. We don't want to
526  * handle reliability issue in this simple test program.
527  */
528  sleep(3);
529 
530  if (message_count) {
531  if (is_sender) {
532  printf("initiating data transfers\n");
533  for (i = 0; i < connections; i++) {
534  ret = post_sends(&test.nodes[i], 0);
535  if (ret)
536  goto out;
537  }
538  ret = poll_cqs();
539  if (ret)
540  goto out;
541  } else {
542  printf("receiving data transfers\n");
543  ret = poll_cqs();
544  if (ret)
545  goto out;
546  }
547  printf("data transfers complete\n");
548  }
549 out:
550  for (i = 0; i < connections; i++) {
551  ret = rdma_leave_multicast(test.nodes[i].cma_id,
552  test.dst_addr);
553  if (ret)
554  perror("mckey: failure leaving");
555  }
556  return ret;
557 }
