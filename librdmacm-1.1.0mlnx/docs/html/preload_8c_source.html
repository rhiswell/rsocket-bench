148 /*
149  * Config file format:
150  * # Starting '#' indicates comment
151  * # wild card values are supported using '*'
152  * # domain - *, INET, INET6, IB
153  * # type - *, STREAM, DGRAM
154  * # protocol - *, TCP, UDP
155  * program_name domain type protocol
156  */
157 static void scan_config(void)
158 {
159  struct config_entry *new_config;
160  FILE *fp;
161  char line[120], prog[64], dom[16], type[16], proto[16];
162 
163  fp = fopen(RS_CONF_DIR "/preload_config", "r");
164  if (!fp)
165  return;
166 
167  while (fgets(line, sizeof(line), fp)) {
168  if (line[0] == '#')
169  continue;
170 
171  if (sscanf(line, "%64s%16s%16s%16s", prog, dom, type, proto) != 4)
172  continue;
173 
174  new_config = realloc(config, (config_cnt + 1) *
175  sizeof(struct config_entry));
176  if (!new_config)
177  break;
178 
179  config = new_config;
180  memset(&config[config_cnt], 0, sizeof(struct config_entry));
181 
182  if (!strcasecmp(dom, "INET") ||
183  !strcasecmp(dom, "AF_INET") ||
184  !strcasecmp(dom, "PF_INET")) {
185  config[config_cnt].domain = AF_INET;
186  } else if (!strcasecmp(dom, "INET6") ||
187  !strcasecmp(dom, "AF_INET6") ||
188  !strcasecmp(dom, "PF_INET6")) {
189  config[config_cnt].domain = AF_INET6;
190  } else if (!strcasecmp(dom, "IB") ||
191  !strcasecmp(dom, "AF_IB") ||
192  !strcasecmp(dom, "PF_IB")) {
193  config[config_cnt].domain = AF_IB;
194  } else if (strcmp(dom, "*")) {
195  continue;
196  }
197 
198  if (!strcasecmp(type, "STREAM") ||
199  !strcasecmp(type, "SOCK_STREAM")) {
200  config[config_cnt].type = SOCK_STREAM;
201  } else if (!strcasecmp(type, "DGRAM") ||
202  !strcasecmp(type, "SOCK_DGRAM")) {
203  config[config_cnt].type = SOCK_DGRAM;
204  } else if (strcmp(type, "*")) {
205  continue;
206  }
207 
208  if (!strcasecmp(proto, "TCP") ||
209  !strcasecmp(proto, "IPPROTO_TCP")) {
210  config[config_cnt].protocol = IPPROTO_TCP;
211  } else if (!strcasecmp(proto, "UDP") ||
212  !strcasecmp(proto, "IPPROTO_UDP")) {
213  config[config_cnt].protocol = IPPROTO_UDP;
214  } else if (strcmp(proto, "*")) {
215  continue;
216  }
217 
218  if (strcmp(prog, "*")) {
219  if (!(config[config_cnt].name = strdup(prog)))
220  continue;
221  }
222 
223  config_cnt++;
224  }
225 
226  fclose(fp);
227  if (config_cnt)
228  atexit(free_config);
229 }

231 static int intercept_socket(int domain, int type, int protocol)
232 {
233  int i;
234 
235  if (!config_cnt)
236  return 1;
237 
238  if (!protocol) {
239  if (type == SOCK_STREAM)
240  protocol = IPPROTO_TCP;
241  else if (type == SOCK_DGRAM)
242  protocol = IPPROTO_UDP;
243  }
244 
245  for (i = 0; i < config_cnt; i++) {
246  if ((!config[i].name ||
247  !strncasecmp(config[i].name, program_invocation_short_name,
248  strlen(config[i].name))) &&
249  (!config[i].domain || config[i].domain == domain) &&
250  (!config[i].type || config[i].type == type) &&
251  (!config[i].protocol || config[i].protocol == protocol))
252  return 1;
253  }
254 
255  return 0;
256 }

258 static int fd_open(void)
259 {
260  struct fd_info *fdi;
261  int ret, index;
262 
263  fdi = calloc(1, sizeof *fdi);
264  if (!fdi)
265  return ERR(ENOMEM);
266 
267  index = open("/dev/null", O_RDONLY);
268  if (index < 0) {
269  ret = index;
270  goto err1;
271  }
272 
273  fdi->dupfd = -1;
274  atomic_init(&fdi->refcnt);
275  atomic_set(&fdi->refcnt, 1);
276  pthread_mutex_lock(&mut);
277  ret = idm_set(&idm, index, fdi);
278  pthread_mutex_unlock(&mut);
279  if (ret < 0)
280  goto err2;
281 
282  return index;
283 
284 err2:
285  real.close(index);
286 err1:
287  free(fdi);
288  return ret;
289 }

341 {

344

451  * We currently only handle copying a few common values.
452  */
453 static int copysockopts(int dfd, int sfd, struct socket_calls *dapi,
454  struct socket_calls *sapi)
455 {
456  socklen_t len;
457  int param, ret;
458 
459  ret = sapi->fcntl(sfd, F_GETFL);
460  if (ret > 0)
461  ret = dapi->fcntl(dfd, F_SETFL, ret);
462  if (ret)
463  return ret;
464 
465  len = sizeof param;
466  ret = sapi->getsockopt(sfd, SOL_SOCKET, SO_REUSEADDR, &param, &len);
467  if (param && !ret)
468  ret = dapi->setsockopt(dfd, SOL_SOCKET, SO_REUSEADDR, &param, len);
469  if (ret)
470  return ret;
471 
472  len = sizeof param;
473  ret = sapi->getsockopt(sfd, IPPROTO_TCP, TCP_NODELAY, &param, &len);
474  if (param && !ret)
475  ret = dapi->setsockopt(dfd, IPPROTO_TCP, TCP_NODELAY, &param, len);
476  if (ret)
477  return ret;
478 
479  return 0;
480 }

482 /*
483  * Convert between an rsocket and a normal socket.
484  */
485 static int transpose_socket(int socket, enum fd_type new_type)
486 {
487  socklen_t len = 0;
488  int sfd, dfd, param, ret;
489  struct socket_calls *sapi, *dapi;
490 
491  sfd = fd_getd(socket);
492  if (new_type == fd_rsocket) {
493  dapi = &rs;
494  sapi = &real;
495  } else {
496  dapi = &real;
497  sapi = &rs;
498  }
499 
500  ret = sapi->getsockname(sfd, NULL, &len);
501  if (ret)
502  return ret;
503 
504  param = (len == sizeof(struct sockaddr_in6)) ? PF_INET6 : PF_INET;
505  dfd = dapi->socket(param, SOCK_STREAM, 0);
506  if (dfd < 0)
507  return dfd;
508 
509  ret = copysockopts(dfd, sfd, dapi, sapi);
510  if (ret)
511  goto err;
512 
513  fd_store(socket, dfd, new_type, fd_ready);
514  return dfd;
515 
516 err:
517  dapi->close(dfd);
518  return ret;
519 }

536 int socket(int domain, int type, int protocol)
537 {
538  static __thread int recursive;
539  int index, ret;
540 
541  init_preload();
542 
543  if (recursive || !intercept_socket(domain, type, protocol))
544  goto real;
545 
546  index = fd_open();
547  if (index < 0)
548  return index;
549 
550  if (fork_support && (domain == PF_INET || domain == PF_INET6) &&
551  (type == SOCK_STREAM) && (!protocol || protocol == IPPROTO_TCP)) {
552  ret = real.socket(domain, type, protocol);
553  if (ret < 0)
554  return ret;
555  fd_store(index, ret, fd_normal, fd_fork);
556  return index;
557  }
558 
559  recursive = 1;
560  ret = rsocket(domain, type, protocol);
561  recursive = 0;
562  if (ret >= 0) {
563  fd_store(index, ret, fd_rsocket, fd_ready);
564  set_rsocket_options(ret);
565  return index;
566  }
567  fd_close(index, &ret);
568 real:
569  return real.socket(domain, type, protocol);
570 }

592 int accept(int socket, struct sockaddr *addr, socklen_t *addrlen)
593 {
594  int fd, index, ret;
595 
596  if (fd_get(socket, &fd) == fd_rsocket) {
597  index = fd_open();
598  if (index < 0)
599  return index;
600 
601  ret = raccept(fd, addr, addrlen);
602  if (ret < 0) {
603  fd_close(index, &fd);
604  return ret;
605  }
606 
607  fd_store(index, ret, fd_rsocket, fd_ready);
608  return index;
609  } else if (fd_gets(socket) == fd_fork_listen) {
610  index = fd_open();
611  if (index < 0)
612  return index;
613 
614  ret = real.accept(fd, addr, addrlen);
615  if (ret < 0) {
616  fd_close(index, &fd);
617  return ret;
618  }
619 
620  fd_store(index, ret, fd_normal, fd_fork_passive);
621  return index;
622  } else {
623  return real.accept(fd, addr, addrlen);
624  }
625 }

627 /*
628  * We can't fork RDMA connections and pass them from the parent to the child
629  * process. Instead, we need to establish the RDMA connection after calling
630  * fork. To do this, we delay establishing the RDMA connection until we try
631  * to send/receive on the server side.
632  */
633 static void fork_active(int socket)
634 {
635  struct sockaddr_storage addr;
636  int sfd, dfd, ret;
637  socklen_t len;
638  uint32_t msg;
639  long flags;
640 
641  sfd = fd_getd(socket);
642 
643  flags = real.fcntl(sfd, F_GETFL);
644  real.fcntl(sfd, F_SETFL, 0);
645  ret = real.recv(sfd, &msg, sizeof msg, MSG_PEEK);
646  real.fcntl(sfd, F_SETFL, flags);
647  if ((ret != sizeof msg) || msg)
648  goto err1;
649 
650  len = sizeof addr;
651  ret = real.getpeername(sfd, (struct sockaddr *) &addr, &len);
652  if (ret)
653  goto err1;
654 
655  dfd = rsocket(addr.ss_family, SOCK_STREAM, 0);
656  if (dfd < 0)
657  goto err1;
658 
659  ret = rconnect(dfd, (struct sockaddr *) &addr, len);
660  if (ret)
661  goto err2;
662 
663  set_rsocket_options(dfd);
664  copysockopts(dfd, sfd, &rs, &real);
665  real.shutdown(sfd, SHUT_RDWR);
666  real.close(sfd);
667  fd_store(socket, dfd, fd_rsocket, fd_ready);
668  return;
669 
670 err2:
671  rclose(dfd);
672 err1:
673  fd_store(socket, sfd, fd_normal, fd_ready);
674 }

676 /*
677  * The server will start listening for the new connection, then send a
678  * message to the active side when the listen is ready. This does leave
679  * fork unsupported in the following case: the server is nonblocking and
680  * calls select/poll waiting to receive data from the client.
681  */
682 static void fork_passive(int socket)
683 {
684  struct sockaddr_in6 sin6;
685  sem_t *sem;

918 static void select_to_rpoll(struct pollfd *fds, int *nfds,
919  fd_set *readfds, fd_set *writefds, fd_set *exceptfds)
920 {

923  for (fd = 0; fd < *nfds; fd++) {
924  events = (readfds && FD_ISSET(fd, readfds)) ? POLLIN : 0;
925  if (writefds && FD_ISSET(fd, writefds))

937 static int rpoll_to_select(struct pollfd *fds, int nfds,
938  fd_set *readfds, fd_set *writefds, fd_set *exceptfds)
939 {
