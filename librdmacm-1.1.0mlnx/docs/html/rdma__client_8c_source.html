47 static int run(void)
48 {
49  struct rdma_addrinfo hints, *res;
50  struct ibv_qp_init_attr attr;
51  struct ibv_wc wc;
52  int ret;
53 
54  memset(&hints, 0, sizeof hints);
55  hints.ai_port_space = RDMA_PS_TCP;
56  ret = rdma_getaddrinfo(server, port, &hints, &res);
57  if (ret) {
58  perror("rdma_getaddrinfo");
59  goto out;
60  }
61 
62  memset(&attr, 0, sizeof attr);
63  attr.cap.max_send_wr = attr.cap.max_recv_wr = 1;
64  attr.cap.max_send_sge = attr.cap.max_recv_sge = 1;
65  attr.cap.max_inline_data = 16;
66  attr.qp_context = id;
67  attr.sq_sig_all = 1;
68  ret = rdma_create_ep(&id, res, NULL, &attr);
69  // Check to see if we got inline data allowed or not
70  if (attr.cap.max_inline_data >= 16)
71  send_flags = IBV_SEND_INLINE;
72  else
73  printf("rdma_client: device doesn't support IBV_SEND_INLINE, "
74  "using sge sends\n");
75 
76  if (ret) {
77  perror("rdma_create_ep");
78  goto out_free_addrinfo;
79  }
80 
81  mr = rdma_reg_msgs(id, recv_msg, 16);
82  if (!mr) {
83  perror("rdma_reg_msgs for recv_msg");
84  ret = -1;
85  goto out_destroy_ep;
86  }
87  if ((send_flags & IBV_SEND_INLINE) == 0) {
88  send_mr = rdma_reg_msgs(id, send_msg, 16);
89  if (!send_mr) {
90  perror("rdma_reg_msgs for send_msg");
91  ret = -1;
92  goto out_dereg_recv;
93  }
94  }
95 
96  ret = rdma_post_recv(id, NULL, recv_msg, 16, mr);
97  if (ret) {
98  perror("rdma_post_recv");
99  goto out_dereg_send;
100  }
101 
102  ret = rdma_connect(id, NULL);
103  if (ret) {
104  perror("rdma_connect");
105  goto out_dereg_send;
106  }
107 
108  ret = rdma_post_send(id, NULL, send_msg, 16, send_mr, send_flags);
109  if (ret) {
110  perror("rdma_post_send");
111  goto out_disconnect;
112  }
113 
114  while ((ret = rdma_get_send_comp(id, &wc)) == 0);
115  if (ret < 0) {
116  perror("rdma_get_send_comp");
117  goto out_disconnect;
118  }
119 
120  while ((ret = rdma_get_recv_comp(id, &wc)) == 0);
121  if (ret < 0)
122  perror("rdma_get_recv_comp");
123  else
124  ret = 0;
125 
126 out_disconnect:
127  rdma_disconnect(id);
128 out_dereg_send:
129  if ((send_flags & IBV_SEND_INLINE) == 0)
130  rdma_dereg_mr(send_mr);
131 out_dereg_recv:
132  rdma_dereg_mr(mr);
133 out_destroy_ep:
134  rdma_destroy_ep(id);
135 out_free_addrinfo:
136  rdma_freeaddrinfo(res);
137 out:
138  return ret;
139 }
