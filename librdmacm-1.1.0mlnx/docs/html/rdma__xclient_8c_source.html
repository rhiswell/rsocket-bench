107  return ret;
108  }
109 
110  if (hints.ai_qp_type == IBV_QPT_XRC_SEND)

112 
113  ret = post_send();

118

123  }
124

131 int main(int argc, char **argv)
132 {
133  int op, ret;
134 
135  hints.ai_port_space = RDMA_PS_TCP;
136  hints.ai_qp_type = IBV_QPT_RC;
137 
138  while ((op = getopt(argc, argv, "s:p:c:")) != -1) {
139  switch (op) {
140  case 's':
141  server = optarg;
142  break;
143  case 'p':
144  strncpy(port, optarg, sizeof port - 1);
145  break;
146  case 'c':
147  switch (tolower(optarg[0])) {
148  case 'r':
149  break;
150  case 'x':
151  hints.ai_port_space = RDMA_PS_IB;
152  hints.ai_qp_type = IBV_QPT_XRC_SEND;
153  break;
154  default:
155  goto err;
156  }
157  break;
158  default:
159  goto err;
160  }
161  }
162 
163  printf("%s: start\n", argv[0]);
164  ret = test();
165  printf("%s: end %d\n", argv[0], ret);
166  return ret;
167 
168 err:

170  printf("\t[-s server]\n");
171  printf("\t[-p port_number]\n");
172  printf("\t[-c communication type]\n");
173  printf("\t r - RC: reliable-connected (default)\n");
174  printf("\t x - XRC: extended-reliable-connected\n");
175  exit(1);
