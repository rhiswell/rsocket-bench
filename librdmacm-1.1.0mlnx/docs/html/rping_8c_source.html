55 /*
56  * rping "ping/pong" loop:
57  * client sends source rkey/addr/len
58  * server receives source rkey/add/len
59  * server rdma reads "ping" data from source
60  * server sends "go ahead" on rdma read completion
61  * client sends sink rkey/addr/len
62  * server receives sink rkey/addr/len
63  * server rdma writes "pong" data to sink
64  * server sends "go ahead" on rdma write completion
65  * <repeat loop>
66  */
67 
68 /*
69  * These states are used to signal events between the completion handler
70  * and the main client or server thread.
71  *
72  * Once CONNECTED, they cycle through RDMA_READ_ADV, RDMA_WRITE_ADV,
73  * and RDMA_WRITE_COMPLETE for each ping.
74  */

170 static int rping_cma_event_handler(struct rdma_cm_id *cma_id,
171  struct rdma_cm_event *event)
172 {
173  int ret = 0;
174  struct rping_cb *cb = cma_id->context;
175 
176  DEBUG_LOG("cma_event type %s cma_id %p (%s)\n",
177  rdma_event_str(event->event), cma_id,
178  (cma_id == cb->cm_id) ? "parent" : "child");
179 
180  switch (event->event) {
181  case RDMA_CM_EVENT_ADDR_RESOLVED:
182  cb->state = ADDR_RESOLVED;
183  ret = rdma_resolve_route(cma_id, 2000);
184  if (ret) {
185  cb->state = ERROR;
186  perror("rdma_resolve_route");
187  sem_post(&cb->sem);
188  }
189  break;
190 
191  case RDMA_CM_EVENT_ROUTE_RESOLVED:
192  cb->state = ROUTE_RESOLVED;
193  sem_post(&cb->sem);
194  break;
195 
196  case RDMA_CM_EVENT_CONNECT_REQUEST:
197  cb->state = CONNECT_REQUEST;
198  cb->child_cm_id = cma_id;
199  DEBUG_LOG("child cma %p\n", cb->child_cm_id);
200  sem_post(&cb->sem);
201  break;
202 
203  case RDMA_CM_EVENT_ESTABLISHED:
204  DEBUG_LOG("ESTABLISHED\n");
205 
206  /*
207  * Server will wake up when first RECV completes.
208  */
209  if (!cb->server) {
210  cb->state = CONNECTED;
211  }
212  sem_post(&cb->sem);
213  break;
214 
215  case RDMA_CM_EVENT_ADDR_ERROR:
216  case RDMA_CM_EVENT_ROUTE_ERROR:
217  case RDMA_CM_EVENT_CONNECT_ERROR:
218  case RDMA_CM_EVENT_UNREACHABLE:
219  case RDMA_CM_EVENT_REJECTED:
220  fprintf(stderr, "cma event %s, error %d\n",
221  rdma_event_str(event->event), event->status);
222  sem_post(&cb->sem);
223  ret = -1;
224  break;
225 
226  case RDMA_CM_EVENT_DISCONNECTED:
227  fprintf(stderr, "%s DISCONNECT EVENT...\n",
228  cb->server ? "server" : "client");
229  cb->state = DISCONNECTED;
230  sem_post(&cb->sem);
231  break;
232 
233  case RDMA_CM_EVENT_DEVICE_REMOVAL:
234  fprintf(stderr, "cma detected device removal!!!!\n");
235  ret = -1;
236  break;
237 
238  default:
239  fprintf(stderr, "unhandled event: %s, ignoring\n",
240  rdma_event_str(event->event));
241  break;
242  }
243 
244  return ret;
245 }

247 static int server_recv(struct rping_cb *cb, struct ibv_exp_wc *wc)
248 {
249  if (wc->byte_len != sizeof(cb->recv_buf)) {
250  fprintf(stderr, "Received bogus data, size %d\n", wc->byte_len);
251  return -1;
252  }
253 
254  cb->remote_rkey = ntohl(cb->recv_buf.rkey);
255  cb->remote_addr = ntohll(cb->recv_buf.buf);
256  cb->remote_len = ntohl(cb->recv_buf.size);
257  DEBUG_LOG("Received rkey %x addr %" PRIx64 " len %d from peer\n",
258  cb->remote_rkey, cb->remote_addr, cb->remote_len);
259 
260  if (cb->state <= CONNECTED || cb->state == RDMA_WRITE_COMPLETE)
261  cb->state = RDMA_READ_ADV;
262  else
263  cb->state = RDMA_WRITE_ADV;
264 
265  return 0;
266 }

283 static int rping_cq_event_handler(struct rping_cb *cb)
284 {
285  struct ibv_exp_wc wc;
286  struct ibv_recv_wr *bad_wr;
287  int ret;
288  int flushed = 0;
289 
290  while ((ret = ibv_exp_poll_cq(cb->cq, 1, &wc, sizeof(wc))) == 1) {
291  ret = 0;
292 
293  if (wc.status) {
294  if (wc.status == IBV_WC_WR_FLUSH_ERR) {
295  flushed = 1;
296  continue;
297 
298  }
299  fprintf(stderr,
300  "cq completion failed status %d\n",
301  wc.status);
302  ret = -1;
303  goto error;
304  }
305 
306  switch (wc.exp_opcode) {
307  case IBV_EXP_WC_SEND:
308  DEBUG_LOG("send completion\n");
309  break;
310 
311  case IBV_EXP_WC_RDMA_WRITE:
312  DEBUG_LOG("rdma write completion\n");
313  cb->state = RDMA_WRITE_COMPLETE;
314  sem_post(&cb->sem);
315  break;
316 
317  case IBV_EXP_WC_RDMA_READ:
318  DEBUG_LOG("rdma read completion\n");
319  cb->state = RDMA_READ_COMPLETE;
320  sem_post(&cb->sem);
321  break;
322 
323  case IBV_EXP_WC_RECV:
324  DEBUG_LOG("recv completion\n");
325  ret = cb->server ? server_recv(cb, &wc) :
326  client_recv(cb, &wc);
327  if (ret) {
328  fprintf(stderr, "recv wc error: %d\n", ret);
329  goto error;
330  }
331 
332  ret = ibv_post_recv(cb->qp, &cb->rq_wr, &bad_wr);
333  if (ret) {
334  fprintf(stderr, "post recv error: %d\n", ret);
335  goto error;
336  }
337  sem_post(&cb->sem);
338  break;
339 
340  case IBV_EXP_WC_BIND_MW:
341  DEBUG_LOG("bind completion\n");
342  break;
343  case IBV_EXP_WC_LOCAL_INV:
344  DEBUG_LOG("local inv completion\n");
345  break;
346 
347  default:
348  DEBUG_LOG("unknown!!!!! completion\n");
349  ret = -1;
350  goto error;
351  }
352  }
353  if (ret) {
354  fprintf(stderr, "poll error %d\n", ret);
355  goto error;
356  }
357  return flushed;
358 
359 error:
360  cb->state = ERROR;
361  sem_post(&cb->sem);
362  return ret;
363 }

409 static int rping_setup_buffers(struct rping_cb *cb)
410 {
411  int ret;
412 
413  DEBUG_LOG("rping_setup_buffers called on cb %p\n", cb);
414 
415  cb->recv_mr = ibv_reg_mr(cb->pd, &cb->recv_buf, sizeof cb->recv_buf,
416  IBV_ACCESS_LOCAL_WRITE);
417  if (!cb->recv_mr) {
418  fprintf(stderr, "recv_buf reg_mr failed\n");
419  return errno;
420  }
421 
422  cb->send_mr = ibv_reg_mr(cb->pd, &cb->send_buf, sizeof cb->send_buf, 0);
423  if (!cb->send_mr) {
424  fprintf(stderr, "send_buf reg_mr failed\n");
425  ret = errno;
426  goto err1;
427  }
428 
429  cb->rdma_buf = malloc(cb->size);
430  if (!cb->rdma_buf) {
431  fprintf(stderr, "rdma_buf malloc failed\n");
432  ret = -ENOMEM;
433  goto err2;
434  }
435 
436  int access = IBV_ACCESS_LOCAL_WRITE |
437  IBV_ACCESS_REMOTE_READ |
438  IBV_ACCESS_REMOTE_WRITE;
439 
440  if (cb->check_mw)
441  access |= IBV_ACCESS_MW_BIND;
442 
443  cb->rdma_mr = ibv_reg_mr(cb->pd, cb->rdma_buf, cb->size, access);
444  if (!cb->rdma_mr) {
445  fprintf(stderr, "rdma_buf reg_mr failed\n");
446  ret = errno;
447  goto err3;
448  }
449 
450  cb->rdma_mw = NULL;
451  if (cb->check_mw) {
452  cb->rdma_mw = ibv_alloc_mw(cb->pd, type_mw);
453  if (!cb->rdma_mw) {
454  fprintf(stderr, "rdma_buf alloc_mw failed\n");
455  ret = -ENOMEM;
456  goto err4;
457  }
458  DEBUG_LOG("rdma_mw rkey initial: %x\n", cb->rdma_mw->rkey);
459  }
460 
461  if (!cb->server) {
462  cb->start_buf = malloc(cb->size);
463  if (!cb->start_buf) {
464  fprintf(stderr, "start_buf malloc failed\n");
465  ret = -ENOMEM;
466  goto err5;
467  }
468 
469  cb->start_mr = ibv_reg_mr(cb->pd, cb->start_buf,
470  cb->size, access);
471  if (!cb->start_mr) {
472  fprintf(stderr, "start_buf reg_mr failed\n");
473  ret = errno;
474  goto err6;
475  }
476  cb->start_mw = NULL;
477  if (cb->check_mw) {
478  cb->start_mw = ibv_alloc_mw(cb->pd, type_mw);
479  if (!cb->start_mw) {
480  fprintf(stderr, "start_buf alloc_mw failed\n");
481  ret = -ENOMEM;
482  goto err7;
483  }
484  DEBUG_LOG("start_mw rkey initial: %x\n",
485  cb->start_mw->rkey);
486  }
487  }
488 
489  rping_setup_wr(cb);
490  DEBUG_LOG("allocated & registered buffers...\n");
491  return 0;
492 
493 err7:
494  ibv_dereg_mr(cb->start_mr);
495 err6:
496  free(cb->start_buf);
497 err5:
498  if (cb->rdma_mw)
499  ibv_dealloc_mw(cb->rdma_mw);
500 err4:
501  ibv_dereg_mr(cb->rdma_mr);
502 err3:
503  free(cb->rdma_buf);
504 err2:
505  ibv_dereg_mr(cb->send_mr);
506 err1:
507  ibv_dereg_mr(cb->recv_mr);
508  return ret;
509 }
