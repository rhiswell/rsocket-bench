83 static int create_message(struct cmatest_node *node)
84 {
85  if (!message_size)
86  message_count = 0;
87 
88  if (!message_count)
89  return 0;
90 
91  node->mem = malloc(message_size + sizeof(struct ibv_grh));
92  if (!node->mem) {
93  printf("failed message allocation\n");
94  return -1;
95  }
96  node->mr = ibv_reg_mr(node->pd, node->mem,
97  message_size + sizeof(struct ibv_grh),
98  IBV_ACCESS_LOCAL_WRITE);
99  if (!node->mr) {
100  printf("failed to reg MR\n");
101  goto err;
102  }
103  return 0;
104 err:
105  free(node->mem);
106  return -1;
107 }

109 static int verify_test_params(struct cmatest_node *node)
110 {
111  struct ibv_port_attr port_attr;
112  int ret;
113 
114  ret = ibv_query_port(node->cma_id->verbs, node->cma_id->port_num,
115  &port_attr);
116  if (ret)
117  return ret;
118 
119  if (message_count && message_size > (1 << (port_attr.active_mtu + 7))) {
120  printf("udaddy: message_size %d is larger than active mtu %d\n",
121  message_size, 1 << (port_attr.active_mtu + 7));
122  return -EINVAL;

128 static int init_node(struct cmatest_node *node)
129 {
130  struct ibv_qp_init_attr init_qp_attr;
131  int cqe, ret;
132 
133  node->pd = ibv_alloc_pd(node->cma_id->verbs);
134  if (!node->pd) {
135  ret = -ENOMEM;
136  printf("udaddy: unable to allocate PD\n");
137  goto out;
138  }
139 
140  cqe = message_count ? message_count * 2 : 2;
141  if (set_ts) {
142  struct ibv_exp_cq_init_attr cq_init_attr;
143  memset(&cq_init_attr, 0, sizeof(cq_init_attr));
144  cq_init_attr.flags = IBV_EXP_CQ_TIMESTAMP;
145  cq_init_attr.comp_mask = IBV_EXP_CQ_INIT_ATTR_FLAGS;
146  node->cq = (struct ibv_cq *)ibv_exp_create_cq(node->cma_id->verbs,
147  cqe, node, NULL, 0,
148  &cq_init_attr);
149  } else
150  node->cq = ibv_create_cq(node->cma_id->verbs, cqe, node, 0, 0);
151  if (!node->cq) {
152  ret = -ENOMEM;
153  printf("udaddy: unable to create CQ\n");
154  goto out;
155  }
156 
157  memset(&init_qp_attr, 0, sizeof init_qp_attr);
158  init_qp_attr.cap.max_send_wr = message_count ? message_count : 1;
159  init_qp_attr.cap.max_recv_wr = message_count ? message_count : 1;
160  init_qp_attr.cap.max_send_sge = 1;
161  init_qp_attr.cap.max_recv_sge = 1;
162  init_qp_attr.qp_context = node;
163  init_qp_attr.sq_sig_all = 0;
164  init_qp_attr.qp_type = IBV_QPT_UD;
165  init_qp_attr.send_cq = node->cq;
166  init_qp_attr.recv_cq = node->cq;
167  ret = rdma_create_qp(node->cma_id, node->pd, &init_qp_attr);
168  if (ret) {
169  perror("udaddy: unable to create QP");
170  goto out;
171  }
172 
173  ret = create_message(node);
174  if (ret) {
175  printf("udaddy: failed to create messages: %d\n", ret);
176  goto out;
177  }
178 out:
179  return ret;
180 }

182 static int post_recvs(struct cmatest_node *node)
183 {
184  struct ibv_recv_wr recv_wr, *recv_failure;
185  struct ibv_sge sge;
186  int i, ret = 0;
187 
188  if (!message_count)
189  return 0;
190 
191  recv_wr.next = NULL;
192  recv_wr.sg_list = &sge;
193  recv_wr.num_sge = 1;
194  recv_wr.wr_id = (uintptr_t) node;
195 
196  sge.length = message_size + sizeof(struct ibv_grh);
197  sge.lkey = node->mr->lkey;
198  sge.addr = (uintptr_t) node->mem;
199 
200  for (i = 0; i < message_count && !ret; i++ ) {
201  ret = ibv_post_recv(node->cma_id->qp, &recv_wr, &recv_failure);
202  if (ret) {
203  printf("failed to post receives: %d\n", ret);
204  break;
205  }
206  }
207  return ret;

210 static int post_sends(struct cmatest_node *node, int signal_flag)
211 {
212  struct ibv_send_wr send_wr, *bad_send_wr;
213  struct ibv_sge sge;
214  int i, ret = 0;
215 
216  if (!node->connected || !message_count)
217  return 0;
218 
219  send_wr.next = NULL;
220  send_wr.sg_list = &sge;
221  send_wr.num_sge = 1;
222  send_wr.opcode = IBV_WR_SEND_WITH_IMM;
223  send_wr.send_flags = signal_flag;
224  send_wr.wr_id = (unsigned long)node;
225  send_wr.imm_data = htonl(node->cma_id->qp->qp_num);
226 
227  send_wr.wr.ud.ah = node->ah;
228  send_wr.wr.ud.remote_qpn = node->remote_qpn;
229  send_wr.wr.ud.remote_qkey = node->remote_qkey;
230 
231  sge.length = message_size;
232  sge.lkey = node->mr->lkey;
233  sge.addr = (uintptr_t) node->mem;
234 
235  for (i = 0; i < message_count && !ret; i++) {
236  ret = ibv_post_send(node->cma_id->qp, &send_wr, &bad_send_wr);
237  if (ret)
238  printf("failed to post sends: %d\n", ret);
239  else if (set_ts) {
240  struct ibv_exp_values queried_values;
241  queried_values.comp_mask = IBV_EXP_VALUES_HW_CLOCK;
242  if (!ibv_exp_query_values(node->cma_id->verbs,
243  IBV_EXP_VALUES_HW_CLOCK,
244  &queried_values) &&
245  (queried_values.comp_mask &
246  IBV_EXP_VALUES_HW_CLOCK)) {
247  printf("Event: Post send was called in %lu\n",
248  queried_values.hwclock);
249  }
250  }
251  }
252  return ret;
253 }

310 static int connect_handler(struct rdma_cm_id *cma_id)
311 {
312  struct cmatest_node *node;
313  struct rdma_conn_param conn_param;
314  int ret;
315 
316  if (test.conn_index == connections) {
317  ret = -ENOMEM;
318  goto err1;
319  }
320  node = &test.nodes[test.conn_index++];
321 
322  node->cma_id = cma_id;
323  cma_id->context = node;
324 
325  ret = verify_test_params(node);
326  if (ret)
327  goto err2;
328 
329  ret = init_node(node);
330  if (ret)
331  goto err2;
332 
333  ret = post_recvs(node);
334  if (ret)
335  goto err2;
336 
337  memset(&conn_param, 0, sizeof conn_param);
338  conn_param.qp_num = node->cma_id->qp->qp_num;
339  ret = rdma_accept(node->cma_id, &conn_param);
340  if (ret) {
341  perror("udaddy: failure accepting");
342  goto err2;
343  }
344  node->connected = 1;
345  test.connects_left--;
346  return 0;
347 
348 err2:
349  node->cma_id = NULL;
350  connect_error();
351 err1:
352  printf("udaddy: failing connection request\n");
353  rdma_reject(cma_id, NULL, 0);
354  return ret;
355 }

376 static int cma_handler(struct rdma_cm_id *cma_id, struct rdma_cm_event *event)
377 {
378  int ret = 0;
379 
380  switch (event->event) {
381  case RDMA_CM_EVENT_ADDR_RESOLVED:
382  ret = addr_handler(cma_id->context);
383  break;
384  case RDMA_CM_EVENT_ROUTE_RESOLVED:
385  ret = route_handler(cma_id->context);
386  break;
387  case RDMA_CM_EVENT_CONNECT_REQUEST:
388  ret = connect_handler(cma_id);
389  break;
390  case RDMA_CM_EVENT_ESTABLISHED:
391  ret = resolved_handler(cma_id->context, event);
392  break;
393  case RDMA_CM_EVENT_ADDR_ERROR:
394  case RDMA_CM_EVENT_ROUTE_ERROR:
395  case RDMA_CM_EVENT_CONNECT_ERROR:
396  case RDMA_CM_EVENT_UNREACHABLE:
397  case RDMA_CM_EVENT_REJECTED:
398  printf("udaddy: event: %s, error: %d\n",
399  rdma_event_str(event->event), event->status);
400  connect_error();
401  ret = event->status;
402  break;
403  case RDMA_CM_EVENT_DEVICE_REMOVAL:
404  /* Cleanup will occur after test completes. */
405  break;
406  default:
407  break;
408  }
409  return ret;
410 }

493 static int poll_cqs(void)
494 {
495  struct ibv_wc wc[8];
496  struct ibv_exp_wc exp_wc[8];
497  int done, i, ret, j;
498  struct ibv_wc *p_wc;
499 
500  for (i = 0; i < connections; i++) {
501  if (!test.nodes[i].connected)
502  continue;
503 
504  for (done = 0; done < message_count; done += ret) {
505  if (set_ts) {
506  ret = ibv_exp_poll_cq(test.nodes[i].cq, 8, exp_wc,
507  sizeof(exp_wc[0]));
508  p_wc = (struct ibv_wc *)exp_wc;
509  } else {
510  ret = ibv_poll_cq(test.nodes[i].cq, 8, wc);
511  p_wc = (struct ibv_wc *)wc;
512  }
513  if (ret < 0) {
514  printf("udaddy: failed polling CQ: %d\n", ret);
515  return ret;
516  }
517  if (set_ts && ret > 0) {
518  struct ibv_exp_values queried_values;
519  queried_values.comp_mask = IBV_EXP_VALUES_HW_CLOCK;
520  if (ibv_exp_query_values(test.nodes[i].cq
521  ->context,
522  IBV_EXP_VALUES_HW_CLOCK,
523  &queried_values) &&
524  (queried_values.comp_mask &
525  IBV_EXP_VALUES_HW_CLOCK)) {
526  printf("Error in querying hw clock\n");
527  queried_values.hwclock = 0;
528  }
529  for (j = 0; j < ret; j++)
530  printf("connection %d, batch %d, "
531  "packet %d/%d %s, "
532  "queried: %lu, timestamp: %lu\n",
533  i, done, j + 1, ret,
534  p_wc->opcode == IBV_WC_SEND ?
535  "tx" : "rx",
536  queried_values.hwclock,
537  exp_wc[j].timestamp);
538  }
539 
540 
541  if (ret && !test.nodes[i].ah)
542  create_reply_ah(&test.nodes[i], p_wc);
543  }
544  }
545  return 0;
546 }

564 {

597  ret = poll_cqs();

599  goto out;
600 
601  printf("sending replies\n");

604  if (ret)

608  ret = poll_cqs();

610  goto out;
611  printf("data transfers complete\n");
612  }
613 out:

615  return ret;

663 int main(int argc, char **argv)
664 {
665  int op, ret;
666 
667  hints.ai_port_space = RDMA_PS_UDP;
668  while ((op = getopt(argc, argv, "s:b:c:C:S:t:p:P:f:T")) != -1) {
669  switch (op) {
670  case 's':
671  dst_addr = optarg;
672  break;
673  case 'b':
674  src_addr = optarg;
675  break;
676  case 'c':
677  connections = atoi(optarg);
678  break;
679  case 'C':
680  message_count = atoi(optarg);
681  break;
682  case 'S':
683  message_size = atoi(optarg);
684  break;
685  case 't':
686  set_tos = 1;
687  tos = (uint8_t) strtoul(optarg, NULL, 0);
688  break;
689  case 'p': /* for backwards compatibility - use -P */
690  hints.ai_port_space = strtol(optarg, NULL, 0);
691  break;
692  case 'f':
693  if (!strncasecmp("ip", optarg, 2)) {
694  hints.ai_flags = RAI_NUMERICHOST;
695  } else if (!strncasecmp("gid", optarg, 3)) {
696  hints.ai_flags = RAI_NUMERICHOST | RAI_FAMILY;
697  hints.ai_family = AF_IB;
698  } else if (strncasecmp("name", optarg, 4)) {
699  fprintf(stderr, "Warning: unknown address format\n");
700  }
701  break;
702  case 'P':
703  if (!strncasecmp("ipoib", optarg, 5)) {
704  hints.ai_port_space = RDMA_PS_IPOIB;
705  } else if (strncasecmp("udp", optarg, 3)) {
706  fprintf(stderr, "Warning: unknown port space format\n");
707  }
708  break;
